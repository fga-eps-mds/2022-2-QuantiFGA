178def separaHorario(horario): 

179 # ========================================================================================================== 

180 # Dado uma string horario no padrão [DIAS_DA_SEMANA][PERIODO][HORARIO] é retornado uma lista do horario  

181 # separado por hora  

182 # [DIAS_DA_SEMANA] 2-7 

183 # [PERIODO] manha = M, tarde = T, e noite = N 

184 # [HORARIO] to do 

185 # exemplo Horario = 26T12 retorno = [[2, "tarde", 1 ], [2, "tarde", 2 ], [6, "tarde", 1 ], [6, "tarde", 2 ]] 

186 # ========================================================================================================== 

187 # separa os horarios se for composto 

188 horarios = horario.split() 

189 resultado = [] 

190 for h in horarios: 

191 # encontra o M ou T ou N 

192 x = re.search("^(\d+)([M|T|N])(\d+)", h) 

193 if x: 

194 # antes do M - T - N estao os dias 

195 days = [int(x) for x in x.group(1)] 

196 #for d in days: 

197 # print(days) 

198 # M - T - N 

199 turno = x.group(2) 

200 #print(turno) 

201 # depois do M - T - N estao as horas 

202 hours = [int(x) for x in x.group(3)] 

203 #for ho in hours: 

204 # print(hours) 

205 # para cada dia e hora cria uma nova linha no dataframe temporario 

206 for d in days: 

207 for ho in hours: 

208 resultado.append(f'{d}{turno}{ho}') 

209 # se nao encontrar M - T - N imprime valores com mensagem para checagem 

210 else: 

211 print("No match", h) 

212 return resultado
